def load_from_env():
    repo_full_name = os.environ.get('MY_REPOSITORY')
    owner = os.environ.get('MY_OWNER')
    repo = repo_full_name[len(owner) + 1:]
    token = os.environ.get('MY_GITHUB_TOKEN')
    github_config = {
        "repo": repo,
        "owner": owner,
        "token": token,
    }
    return github_config
